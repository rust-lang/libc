    #[allow(missing_debug_implementations)]
    #[repr(align(16))]
    pub struct max_align_t {
        priv_: [f64; 4]
    }

    #[repr(align(16))]
    pub struct mcontext_t {
        pub mc_onstack: register_t,

        pub mc_rsi: register_t,
        pub mc_rdx: register_t,
